git小笔记

1. 下载windows下安装的git安装包
	下载地址：https://git-scm.com/download/win
	下载好后，双击安装
	
2. 配置用户信息
	// 配置全局用户名
	git config --global user.name "用户名"
	// 配置全局邮箱
	git config --global user.email "邮箱(注册GitHub的邮箱)"
	// 查看配置是否成功
	git config --list
	
	Sourcetree：git的一个图形化工具
	
3. git本地仓库的创建
	使用gitbash进入到一个空文件夹中
	// 初始化版本库
	git init
	
	在本地版本库中创建一个文件
	// 将文件添加到暂存区
	git add 文件名
	// 提交到本地库，-m：对应此次提交的描述信息，建议每次提交都跟上描述信息，方便回滚或者对文件的查找
	git commit -m "描述信息"
	// 查看当前本地库的状态(查看是否有更新的文件但没有提交/文件添加到暂存区但是没有提交到本地库等等)
	git status
	
	// 查看所在位置
	pwd
	// 查看当前文件夹中文件信息
	ll
	// 移动到上一层目录
	cd ..
	// 移动到某个文件夹中
	cd 文件夹名称
	// 创建文件夹
	mkdir 名称
	/ /查看隐藏文件
	ls -a
	// 添加文本追加到文件里面，">>" 是管道追加的意思
	echo "文本" >> 文件名.后缀
	// 展示文件内容
	cat 文件名
